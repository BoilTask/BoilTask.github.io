#include<stdio.h>
#include<algorithm>
using namespace std;
int map[352][352];
int main() {
    int n;
    while(scanf("%d",&n)!=EOF) {
        for(int i=0; i<n; i++)
            for(int j=0; j<=i; j++)
                scanf("%d",&map[i][j]);
        if(n==1) {
            printf("%d\n",map[0][0]);
            continue;
        }
        for(int i=n-2; i>=0; i--) {
            for(int j=0; j<=i; j++) {
                map[i][j]+=max(map[i+1][j],map[i+1][j+1]);
            }
        }
        printf("%d\n",map[0][0]);
    }
    return 0;
}
